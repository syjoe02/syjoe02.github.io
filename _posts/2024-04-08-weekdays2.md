# Single Linekd List

### 1. Code
```c
#include <stdio.h>
#include <stdlib.h>

// define the node structure
struct node {
    int num;
  struct node *next;
};

// create a new node
struct node* createNode(int num) {
  struct node* newNode = (struct node*)malloc(sizeof(struct node));
  newNode->num = num;
  newNode->next = NULL;
  return newNode;
}

// to append a node to the end of the linked list
void appendNode(struct node** head, int num) {
  struct node* newNode = createNode(num);
  if (*head == NULL) {
    *head = newNode;
  } else {
    struct node* temp = *head;
    while (temp->next != NULL) {
      temp = temp->next;
    }
    temp->next = newNode;
  }
}

// to traverse the linked list and print each node's data
void printList(struct node* head) {
  struct node* temp = head;
  while (temp != NULL) {
    printf("%d -> ", temp->num);
    temp = temp->next;
  }
  printf("NULL\n");
}

int main() {
  struct node* head = NULL;  // Head pointer (the start of the linked list)

  appendNode(&head, 1);  // Add nodes to the list
  appendNode(&head, 2);
  appendNode(&head, 3);

  printList(head);

  // deallocation
  while (head != NULL) {
    struct node* temp = head;
    head = head->next;
    free(temp);
  }

  return 0;
}
```
